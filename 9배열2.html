배열조작

배열합치기(concat)

        let str = '대한'.concat('민국');
        console.log(str);// 대한민국
        
        let first = '대한';
        let last = '민국';
        let full = first.concat(last);
        console.log(full);// 대한민국

        let arr1 = [1,2,3];
        let arr2 = [4,5,6];
        let addArr = arr1.concat(arr2);
        console.log(`addArr = ${addArr}`);//[1,2,3,4,5,6]

배열나누기(slice)

        const arr3 = ['오렌지','사과','배','참외','수박'];
        const arr4 = arr3.slice(1,3);//slice(시작인덱스,끝인덱스+1);
        console.log(arr4);
        const arr5 = arr3.slice(-3,-1);
        console.log(arr5);

위치 찾기(indexOf)

        let arr6 = ['오렌지','사과','배','참외','수박'];
        let idxNum = arr6.indexOf('배');
        console.log(`배의 위치는 arr6에서 ${idxNum}번째 입니다.`);

        let subArr1 = arr6.slice(0,arr6.indexOf('배'));
        let subArr2 = arr6.slice(arr6.indexOf('배'),arr6.length);
        console.log(`subArr1 = ${subArr1}`);
        console.log(`subArr2 = ${subArr2}`);

정렬(오름차순,역순정렬)

        오름차순정렬
        let arr7 = ['오렌지','사과','배','참외','수박'];
        arr7.sort();
        console.log(`arr7을 오름차순 정렬하니 "${arr7}"가 되었다`);
        역순정렬
        arr7.reverse();
        console.log(`arr7을 내림차순 정렬하니 "${arr7}"가 되었다`);

배열(개체) 복사

        let arr8 = ['오렌지','사과','배','참외','수박'];
        let arr9 = arr8;//copy X, 참조복사
        console.log(`arr8=${arr8},arr9=${arr9}`);
        arr9[0]='앵두';
        console.log(`arr8=${arr8},arr9=${arr9}`);

        let arr10 = ['오렌지','사과','배','참외','수박'];
        let arr11 = [...arr10];//스프레드(spread)연산자
        console.log(`arr10=${arr10},arr11=${arr11}`);
        arr11[0]='앵두';
        console.log(`arr10=${arr10},arr11=${arr11}`);
