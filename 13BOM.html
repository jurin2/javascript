navigator Object

        // navigator Object(브라우저 정보)
        let 브라우저이름 = navigator.appName;
        console.log(`브라우저이름 : ${브라우저이름}`);
        let 브라우저코드이름 = navigator.appCodeName;

        let 브라우저버전 = navigator.appVersion;
