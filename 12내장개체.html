수학

        //수학개체
        let pi = Math.PI;
        let r = 5;
        let width = Math.pow(r,2)*pi;
        console.log(width);
        console.log(Math.abs(-5));//절대값
        console.log(Math.max(-5,4,67,12,98));//최대값
        console.log(Math.min(-5,4,67,12,98));//최소값
        console.log(Math.round(256.4289,2));//반올림
        console.log(Math.ceil(256.4289));//올림
        console.log(Math.floor(256.4289));//내림
        console.log(Math.sqrt(4));//루트
        console.log(Math.pow(4,3));//제곱값
        console.log(Math.random());//랜덤값
        console.log(Math.PI);//원주율값
